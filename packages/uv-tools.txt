# uv-managed tools. Consumed by install/install-tools.sh (needs `uv`). Two forms:
#   install <package>            → `uv tool install <package>` (persistent CLI on PATH)
#   uvx <name> <command...>      → writes ~/.local/bin/<name> that execs <command> "$@"
#                                  (a stable wrapper around an ephemeral uvx invocation)
# Extend freely — this is the source of truth for uv-based tooling.

# --- uv tool install (persistent) ---
install ruff
install ty
install aider-chat --python 3.12   # pin: numpy has no wheel on newer pythons → avoids source build (needs cc)

# --- uvx wrappers (ephemeral, wrapped as ~/.local/bin/<name>) ---
uvx fast-agent-mcp uvx fast-agent-mcp
